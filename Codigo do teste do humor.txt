#include <stdio.h>
#include <string.h>
#include <locale.h>

#include <stdio.h>

int obterResposta(const char *pergunta) {
    int resposta;
    do {
        printf("%s (0-10): ", pergunta);
        scanf("%d", &resposta);
    } while (resposta < 0 || resposta > 10);
    return resposta;
}
